Примеры работы плагина модального окна Wumodal.

1.Простой вызов: 
$('.call-1').wumodal();
2.Вызов с пользовательскими параметрами: 
$('.call-2').wumodal({
	opacityOverlay : 0.9, // прозрачность затемнения 90%
	modalWidth : 300, // ширина окна
	closeClickOverlay : false, // отменяем закрытие окна при клике на затемнение
	positionContainer : 'bottom', // с какой стороны будет появляется окно
	containerClass : 'newclass', // доп. класс для стилизации этого окна
	template: '<p>Базовый шаблон легко изменить, написать здесь текст или вставить форму </p>' + 
	'<input type="text">' + '<input type="button" value="Готово">'
});
3.Вызов по hover + параметры: 
$('.call-3').wumodal({
	activeEvent : 'hover', // так же можно указать значение mouseover
	positionContainer : 'center',	// c какой стороны будет появляться
});
4.Вызов темы dark + параметры 
$('.call-4').wumodal({
	positionContainer : 'left', // c какой стороны будет появляется
	theme : 'dark', // выберем тему dark
	template: 'Используем для этого модального окна тему dark'// изменим стандартный шаблон 
});
5.Вызов темы fancybox + параметры 
$('.call-5').wumodal({
	speedModal : 800, // Скорость появления окна
	modalHeight: 200, // Фиксированная высота окна
	theme : 'fancybox',// выберем тему fancybox
	template: 'Используем для этого модального окна тему fancybox',// изменим стандартный шаблон
});
6.Вызов темы lightgreen 
$('.call-6').wumodal({
	overlayBg : '#0F3547',// цвет затемнения
	theme : 'lightgreen',// выберем тему lightgreen
	template : 'Используем для этого модального окна тему lightgreen', // изменим стандартный шаблон
});
7.Вызов темы lightred 
$('.call-7').wumodal({
	modalWidth : 300, // фиксированная ширина для этого окна
	theme : 'lightred', // выберем тему lightred
	template : '<h2 class="head-modal">Модальное окно</h2>'+'
	<div class="inner-content">"Lorem ipsum dolor sit amet, consectetur adipisicing elit, sed do eiusmod 
tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco 
laboris nisi ut aliquip ex ea commodo consequat. Duis aute irure dolor in reprehenderit in voluptate velit 
esse cillum dolore eu fugiat nulla pariatur."</div>', // более сложный шаблон
});
