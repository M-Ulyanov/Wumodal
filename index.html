Примеры работы плагина модального окна Wumodal.

Простой вызов 
$('.call-1').wumodal();
Вызов с пользовательскими параметрами 
$('.call-2').wumodal({
	opacityOverlay : 0.9, //прозрачность затемнения
	modalWidth : 300, //ширина окна
	closeClickOverlay : false, //отменяем закрытие окна при клике на затемнение
	positionContainer : 'bottom', // с какой стороны будет появляется окно
	containerClass : 'newclass', // доп. класс для стилизации этого окна
	template: '<p>Базовый шаблон легко изменить, написать здесь текст или вставить форму </p>' + 
	'<input type="text">' + '<input type="button" value="Готово">'
});
Вызов по hover + параметры 
$('.call-3').wumodal({
	activeEvent : 'hover', //так же можно указать значение mouseover
	positionContainer : 'center',	// c какой стороны будет появляться. 
});
